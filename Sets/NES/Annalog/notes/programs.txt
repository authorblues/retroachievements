Opcodes
    Empty = 0x00  ; (no args)        Jumps to program start (in zero cycles).
    Copy  = 0x01  ; r, r/i           Sets register to value of second arg.
    Sync  = 0x02  ; (no args)        Pauses until all machines are at a SYNC
    Add   = 0x03  ; r, r/i, r/i      Sets register to sum of other args.
    Sub   = 0x04  ; r, r/i, r/i      Sets register to difference of other args.
    Mul   = 0x05  ; r, r/i, r/i      Sets register to product of other args.
    Goto  = 0x06  ; addr             Jumps to an absolute address (0-15).
    Skip  = 0x07  ; r/i              Skips next N instructions.
    If    = 0x08  ; oper, r/i, r/i   Skips next instruction if false.
    Til   = 0x09  ; oper, r/i, r/i   Jumps to prev instruction if false.
    Act   = 0x0a  ; (no args)        Performs machine action (e.g. firing gun).
    Move  = 0x0b  ; dir              Moves machine in the specified direction.
    Rest  = 0x0c  ; r/i              Pauses for N * 16 frames.
    Beep  = 0x0d  ; r/i              Plays a sound with one of ten pitches.
    End   = 0x0e  ; (no args)        Halts the program.
    Nop   = 0x0f  ; (no args)        Does nothing (but still takes a cycle).

Comparison
    Eq = 0 ; equal
    Ne = 1  ; not equal
    Lt = 2  ; less than
    Le = 3  ; less than or equal
    Gt = 4  ; greater than
    Ge = 5  ; greater than or equal

Direction
    Up    = 0
    Right = 1
    Down  = 2
    Left  = 3

Instruction Fields
    Erase     = 0 ; used for the erase-program menu (not really a field type)
    Debug     = 1 ; used for the debug menu (not really a field type)
    Opcode    = 2 ; the instruction's opcode
    LValue    = 3 ; a writable register
    RValue    = 4 ; an immediate value (0-9) or readable register
    Address   = 5 ; an instruction number (for GOTO instructions)
    Compare   = 6 ; a comparison operator (for IF and TIL instructions)
    Direction = 7 ; a cardinal direction (for MOVE instructions)

program instructions are 2 bytes
    ;; The low nibble stores the second argument (if any), and the high nibble
    ;; stores the third argument (if any).
    Arg_byte .byte
    ;; The high nibble stores an eOpcode, and the low nibble stores the first
    ;; argument (if any).
    Op_byte  .byte
programs are 16 instructions
    ;; The instructions in the program.  The length of the program is the index
    ;; of the first eOpcode::Empty instruction if any, or kMaxProgramLength
    ;; otherwise.
    Code_sIns_arr .res .sizeof(sIns) * kMaxProgramLength
